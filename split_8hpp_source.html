95 using tuple_t = ::beman::execution26::detail::decayed_tuple<Tag, Args...>;
96 try {

124 template <class Sndr>
125 struct shared_state {
126 template <class... Args>
127 using value_tuple = ::std::tuple<::beman::execution26::set_value_t, ::std::decay_t<Args>...>;
128
129 template <class... Args>
130 using error_tuples = ::std::variant<::std::tuple<::beman::execution26::set_error_t, ::std::decay_t<Args>>...>;
131
132 using variant_type = ::beman::execution26::detail::meta::unique<::beman::execution26::detail::meta::combine<
133 ::std::variant<::std::monostate>,
134 ::std::variant<::std::tuple<::beman::execution26::set_stopped_t>>,
135 ::std::variant<::std::tuple<::beman::execution26::set_error_t, ::std::exception_ptr>>,
136 ::beman::execution26::error_types_of_t<Sndr, split_env, error_tuples>,
137 ::beman::execution26::value_types_of_t<Sndr, split_env, value_tuple>>>;
138
139 using state_list_type = ::beman::execution26::detail::atomic_intrusive_stack<&local_state_base::next>;
140
141 using child_operation_state = ::beman::execution26::connect_result_t<Sndr, split_receiver<Sndr>>;
142
143 explicit shared_state(Sndr&& sndr) {
144 try {
145 op_state.emplace(::beman::execution26::detail::emplace_from{[&] {
146 return ::beman::execution26::connect(::std::forward<Sndr>(sndr), split_receiver<Sndr>{this});
147 }});
148 } catch (...) {
149 using error_tuple_t = ::std::tuple<::beman::execution26::set_error_t, ::std::exception_ptr>;
150 result.template emplace<error_tuple_t>(::beman::execution26::set_error, ::std::current_exception());
151 [[maybe_unused]] auto queue = waiting_states.pop_all_and_shutdown();
152 assert(queue.empty());
153 }
154 }
155
156 // We use an intrusive list to store the listeners that are waiting for the operation to complete.
157 // if the intrusive list is empty, we start the operation
158 // if the intrusive list is not empty, we push the listener to the intrusive list
159 // if the intrusive list is shutdown, we immediately notify the listener
160 void add_listener(local_state_base* listener) noexcept {
161 // try to push the listener to the intrusive list, if the intrusive list is empty and not shutdown, start
162 // the operation
163 if (auto maybe_ptr = waiting_states.try_push(listener); !maybe_ptr) {
164 // the queue is shutdown, immediately notify the listener
165 listener->notify();
166 } else if (!(*maybe_ptr)) {
167 // the operation was not started yet, we are first, and we start it
168 assert(op_state);
169 ::beman::execution26::start(*op_state);
170 }
171 }
172
173 void notify() noexcept {
174 // note: this is different from stdexec.
175 // we discussed lifetime durations of operation at LEWG and haven't decided yet
176 // whether we should keep the operation alive as long as possible
177 op_state.reset();
178 auto listeners = waiting_states.pop_all_and_shutdown();
179 while (auto listener = listeners.pop()) {
180 listener->notify();
181 }
182 }
183
184 void inc_ref() noexcept { ref_count.fetch_add(1); }
185
186 // This is the most complicated part of the split implementation.
187 // On construction, the operation state increments the ref count.
188 // Before the operation is started, at least one listener is added to the queue.
189 // If the ref count is decreased to one and the there are no listeners in the queue
190 // the operation state is the last object holding the shared state and we can safely
191 // destroy it
192 //
193 // it is not thread safe to destroy a split-sender and copy it at the same time
194 // this is similar to how a shared_ptr is not thread safe to copy and destroy at the same time
195 void dec_ref() noexcept {
196 std::size_t count = ref_count.load();
197 if (count == 2 && waiting_states.empty_and_not_shutdown()) {
198 assert(op_state);
199 [[maybe_unused]] auto listeners = waiting_states.pop_all_and_shutdown();
200 assert(listeners.empty());
201 op_state.reset();
202 }
203 if (ref_count.fetch_sub(1) == 1) {
204 delete this;
205 }
206 }
207
208 ::beman::execution26::inplace_stop_source stop_src{};
209 variant_type result{};
210 state_list_type waiting_states{};
211 ::std::atomic<::std::size_t> ref_count{0};
212 ::std::optional<child_operation_state> op_state{};
213 };

215 template <class Sndr, class Receiver>
216 struct local_state final : local_state_base {
217 using stop_token_type = ::beman::execution26::stop_token_of_t<::beman::execution26::env_of_t<Receiver>>;
218
219 struct on_stop_type {
220 shared_state<Sndr>* sh_state;
221 void operator()() noexcept { sh_state->stop_src.request_stop(); }
222 };
223
224 using on_stop_callback = ::beman::execution26::stop_callback_for_t<stop_token_type, on_stop_type>;
225
226 explicit local_state(shared_state<Sndr>* state, Receiver& rcvr) noexcept
227 : sh_state(state), receiver{std::addressof(rcvr)} {
228 sh_state->inc_ref();
229 }
230
231 ~local_state() noexcept { sh_state->dec_ref(); }
232
233 local_state(const local_state&) = delete;
234 local_state& operator=(const local_state&) = delete;
235 local_state(local_state&&) = delete;
236 local_state& operator=(local_state&&) = delete;
237
238 auto notify() noexcept -> void override {
239 on_stop.reset();
240 auto stop_token = ::beman::execution26::get_stop_token(::beman::execution26::get_env(receiver));
241 if (stop_token.stop_requested()) {
242 ::beman::execution26::set_stopped(std::move(*receiver));
243 } else {
244 assert(sh_state->result.index() > 0);
245 assert(!sh_state->result.valueless_by_exception());
246 try {
247 ::std::visit(
248 [&]<class Arg>(const Arg& arg) noexcept -> void {
249 if constexpr (not ::std::same_as<::std::decay_t<Arg>, ::std::monostate>) {
250 ::std::apply(
251 [&](auto tag, const auto&... args) noexcept -> void {
252 tag(::std::move(*receiver), args...);
253 },
254 arg);
255 }
256 },
257 sh_state->result);
258 } catch (...) {
259 // required by clang-tidy although it is not necessary here
260 // see valueless_by_exception() check above
261 std::terminate();
262 }
263 }
264 }
265
266 void start() noexcept {
267 on_stop.emplace(::beman::execution26::get_stop_token(::beman::execution26::get_env(*receiver)),
268 on_stop_type{sh_state});
269 sh_state->add_listener(this);
270 }
271
272 std::optional<on_stop_callback> on_stop;
273 shared_state<Sndr>* sh_state;
274 Receiver* receiver;
275 };

311 // check for self-assignment was required by clang-tidy
312 // although it is not necessary here

314 return *this;

317 ::std::swap(sh_state, tmp.sh_state);
318 return *this;
319 }

322 auto tmp = ::std::move(other);
323 ::std::swap(sh_state, tmp.sh_state);
324 return *this;
