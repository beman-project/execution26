11// std::forward_like() doesn't work on some compilers, yet. This header
12// provides a work-around.

14namespace beman::execution26::detail {
15template <typename>
16struct forward_like_helper;
17
18template <typename T>
19struct forward_like_helper {
20 template <typename U>
21 static auto forward(U&& u) -> ::std::remove_reference_t<U>&& {
22 return ::std::move(u); // NOLINT(bugprone-move-forwarding-reference)
23 }
24};
25template <typename T>
26struct forward_like_helper<T&&> {
27 template <typename U>
28 static auto forward(U&& u) -> ::std::remove_cvref_t<U>&& {
29 return ::std::move(u); // NOLINT(bugprone-move-forwarding-reference)
30 }
31};
32template <typename T>
33struct forward_like_helper<T&> {
34 template <typename U>
35 static auto forward(U&& u) -> ::std::remove_cvref_t<U>& {
36 return ::std::forward<U&&>(u);
37 }
38};
39template <typename T>
40struct forward_like_helper<const T&&> {
41 template <typename U>
42 static auto forward(U&& u) -> const ::std::remove_cvref_t<U>&& {
43 return ::std::move(u); // NOLINT(bugprone-move-forwarding-reference)
44 }
45};
46template <typename T>
47struct forward_like_helper<const T&> {
48 template <typename U>
49 static auto forward(U&& u) -> const ::std::remove_cvref_t<U>& {
50 return ::std::forward<U&&>(u);
51 }
52};

54// The overload own_forward_like is used for testing on systems
55// which actually do provide an implementation.
56template <typename T, typename U>
57auto own_forward_like(U&& u) noexcept -> decltype(auto) {
58 return ::beman::execution26::detail::forward_like_helper<T>::forward(::std::forward<U>(u));
59}

61template <typename T, typename U>
62auto forward_like(U&& u) noexcept -> decltype(auto) {
63#if 202207 <= disabled__cpp_lib_forward_like
64 return ::std::forward_like<T>(::std::forward<U>(u));
65#else
66 return ::beman::execution26::detail::forward_like_helper<T>::forward(::std::forward<U>(u));
67#endif
68}
