166 template <typename Receiver, typename... Args>
167 static auto let_bind(auto& state, Receiver& receiver, Args&&... args) {
168 using args_t = ::beman::execution26::detail::decayed_tuple<Args...>;
169 auto mkop{[&] {
170 return ::beman::execution26::connect(
171 ::std::apply(::std::move(state.fun),
172 state.args.template emplace<args_t>(::std::forward<Args>(args)...)),
173 let_receiver<Receiver, decltype(state.env)>{receiver, state.env});
174 }};
175 ::beman::execution26::start(
176 state.ops2.template emplace<decltype(mkop())>(beman::execution26::detail::emplace_from{mkop}));
177 }
178 static constexpr auto complete{
179 []<class Tag, class... Args>(auto, auto& state, auto& receiver, Tag, Args&&... args) {
180 if constexpr (::std::same_as<Tag, Completion>) {
181 try {
182 let_bind(state, receiver, ::std::forward<Args>(args)...);
183 } catch (...) {
184 ::beman::execution26::set_error(::std::move(receiver), ::std::current_exception());
185 }
186 } else {
