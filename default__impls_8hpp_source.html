23struct default_impls {
24 static constexpr auto get_attrs = [](const auto&, const auto&... child) noexcept -> decltype(auto) {
25 if constexpr (1 == sizeof...(child))
26 return (::beman::execution26::detail::fwd_env(::beman::execution26::get_env(child)), ...);
27 else
28 return ::beman::execution26::empty_env{};
29 };
30 static constexpr auto get_env = [](auto, auto&, const auto& receiver) noexcept -> decltype(auto) {
31 return ::beman::execution26::detail::fwd_env(::beman::execution26::get_env(receiver));
32 };
33 static constexpr auto get_state =
34 []<typename Sender, typename Receiver>(Sender&& sender, Receiver& receiver) noexcept -> decltype(auto) {
35 auto&& decompose = ::beman::execution26::detail::get_sender_data(::std::forward<Sender>(sender));
36
37 return ::beman::execution26::detail::allocator_aware_move(
38 ::beman::execution26::detail::forward_like<Sender>(decompose.data), receiver);
39 };
40 static constexpr auto start = [](auto&, auto&, auto&... ops) noexcept -> void {
41 (::beman::execution26::start(ops), ...);
42 };
43 static constexpr auto complete = []<typename Index, typename Receiver, typename Tag, typename... Args>(
44 Index, auto&, Receiver& receiver, Tag, Args&&... args) noexcept -> void
45 requires ::beman::execution26::detail::callable<Tag, Receiver, Args...>
46 {
47 static_assert(Index::value == 0);
48 Tag()(::std::move(receiver), ::std::forward<Args>(args)...);
49 };
