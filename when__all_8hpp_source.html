119 void arrive(Receiver& recvr) noexcept {
120 if (0u == --count)
121 this->complete(recvr);

124 void complete(Receiver& recvr) noexcept {
125 switch (this->disp) {
126 case disposition::started: {
127 auto tie = []<typename... T>(::std::tuple<T...>& t) noexcept {
128 return ::std::apply([](auto&... a) { return ::std::tie(a...); }, t);
129 };
130 auto set = [&](auto&... t) noexcept {
131 ::beman::execution26::set_value(::std::move(recvr), ::std::move(t)...);
132 };
133
134 this->on_stop.reset();
135 ::std::apply([&](auto&... opts) noexcept { ::std::apply(set, ::std::tuple_cat(tie(*opts)...)); },
136 this->values);
137 } break;
138 case disposition::error:
139 this->on_stop.reset();
140 try {
141 ::std::visit(
142 [&]<typename Error>(Error& error) noexcept {
143 if constexpr (!::std::same_as<Error, nonesuch>) {
144 ::beman::execution26::set_error(::std::move(recvr), ::std::move(error));
145 }
146 },
147 this->errors);
148 } catch (...) {
149 ::beman::execution26::set_error(::std::move(recvr), ::std::current_exception());
150 }
151 break;
152 case disposition::stopped:
153 this->on_stop.reset();
154 ::beman::execution26::set_stopped(::std::move(recvr));
155 break;
156 }
157 }

159 auto request_stop() -> void {
160 if (1u == ++this->count)
161 --this->count;
162 else {
163 this->stop_src.request_stop();
164 this->arrive(*this->receiver);
165 }
166 }

200 static constexpr auto complete{
201 []<typename Index, typename State, typename Receiver, typename Set, typename... Args>(
202 Index, State& state, Receiver& receiver, Set, Args&&... args) noexcept -> void {
203 if constexpr (::std::same_as<Set, ::beman::execution26::set_error_t>) {
204 if (disposition::error != state.disp.exchange(disposition::error)) {
205 state.stop_src.request_stop();
206 try {
207 state.errors.template emplace<typename ::std::decay<Args...>::type>(
208 ::std::forward<Args>(args)...);
209 } catch (...) {
210 state.errors.template emplace<::std::exception_ptr>(::std::current_exception());
211 }
212 }
213 } else if constexpr (::std::same_as<Set, ::beman::execution26::set_stopped_t>) {
214 auto expected = disposition::started;
215 if (state.disp.compare_exchange_strong(expected, disposition::stopped)) {
216 state.stop_src.request_stop();
217 }
218 } else if constexpr (!::std::same_as<decltype(State::values), ::std::tuple<>>) {
219 if (state.disp == disposition::started) {
220 auto& opt = ::std::get<Index::value>(state.values);
221 try {
222 opt.emplace(::std::forward<Args>(args)...);
223 } catch (...) {
224 if (disposition::error != state.disp.exchange(disposition::error)) {
225 state.stop_src.request_stop();
226 state.errors.template emplace<::std::exception_ptr>(::std::current_exception());
227 }
228 }
229 }
230 }
231 state.arrive(receiver);
232 }};
