41template <typename Awaiter, ::beman::execution26::receiver Receiver>
42auto connect_awaitable(Awaiter awaiter, Receiver receiver)
43 -> ::beman::execution26::detail::operation_state_task<Receiver>
44 requires ::beman::execution26::
45 receiver_of<Receiver, ::beman::execution26::detail::awaiter_completion_signatures<Awaiter, Receiver> >
46{
47 // NOTE: suspened_complete(...) is co_await to make sure that the
48 // coroutine is suspended at the point when set_*(...) is called.
49 using result_type = ::beman::execution26::detail::
50 await_result_type<Awaiter, ::beman::execution26::detail::connect_awaitable_promise<Receiver> >;
51
52 ::std::exception_ptr ep;
53 try {
54 if constexpr (std::same_as<void, result_type>) {
55 co_await ::std::move(awaiter);
56 co_await ::beman::execution26::detail::suspend_complete(::beman::execution26::set_value,
57 ::std::move(receiver));
58 } else {
59 co_await ::beman::execution26::detail::suspend_complete(
60 ::beman::execution26::set_value, ::std::move(receiver), co_await ::std::move(awaiter));
61 }
62 } catch (...) {
63 ep = ::std::current_exception();
64 }
65 co_await ::beman::execution26::detail::suspend_complete(
66 ::beman::execution26::set_error, ::std::move(receiver), ep);
67}
