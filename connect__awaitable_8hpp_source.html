61template <typename Awaiter, ::beman::execution26::receiver Receiver>
62auto connect_awaitable(Awaiter awaiter, Receiver receiver)
63 -> ::beman::execution26::detail::operation_state_task<Receiver>
64 requires ::beman::execution26::
65 receiver_of<Receiver, ::beman::execution26::detail::awaiter_completion_signatures<Awaiter, Receiver> >
66{
67 // NOTE: suspened_complete(...) is co_await to make sure that the
68 // coroutine is suspended at the point when set_*(...) is called.
69 using result_type = ::beman::execution26::detail::
70 await_result_type<Awaiter, ::beman::execution26::detail::connect_awaitable_promise<Receiver> >;
71
72 ::std::exception_ptr ep;
73 try {
74 if constexpr (std::same_as<void, result_type>) {
75 co_await ::std::move(awaiter);
76 co_await ::beman::execution26::detail::suspend_complete(::beman::execution26::set_value,
77 ::std::move(receiver));
78 } else {
79 co_await ::beman::execution26::detail::suspend_complete(
80 ::beman::execution26::set_value, ::std::move(receiver), co_await ::std::move(awaiter));
81 }
82 } catch (...) {
83 ep = ::std::current_exception();
84 }
85 co_await ::beman::execution26::detail::suspend_complete(
86 ::beman::execution26::set_error, ::std::move(receiver), ep);
87}
