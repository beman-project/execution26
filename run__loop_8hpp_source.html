91 enum class state : unsigned char { starting, running, finishing };
92
93 state current_state{state::starting};
94 ::std::mutex mutex{};
95 ::std::condition_variable condition{};
96 opstate_base* front{};
97 opstate_base* back{};
98
99 auto push_back(opstate_base* item) -> void {
100 ::std::lock_guard guard(this->mutex);
101 if (auto previous_back{::std::exchange(this->back, item)}) {
102 previous_back->next = item;
103 } else {
104 this->front = item;
105 this->condition.notify_one();
106 }
107 }
108 auto pop_front() -> opstate_base* {
109 ::std::unique_lock guard(this->mutex);
110 this->condition.wait(guard, [this] { return this->front || this->current_state == state::finishing; });
111 if (this->front == this->back)
112 this->back = nullptr;
113 return this->front ? ::std::exchange(this->front, this->front->next) : nullptr;
114 }

130 auto run() -> void {
131 if (::std::lock_guard guard(this->mutex);
132 this->current_state != state::finishing &&
133 state::running == ::std::exchange(this->current_state, state::running)) {
134 ::std::terminate();
135 }
136
137 while (auto* op{this->pop_front()}) {
138 op->execute();
139 }
140 }
