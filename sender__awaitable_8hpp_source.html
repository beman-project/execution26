28template <class Sndr, class Promise>
29class sender_awaitable {
30 struct unit {};
31 using value_type =
32 ::beman::execution26::detail::single_sender_value_type<Sndr, ::beman::execution26::env_of_t<Promise>>;
33 using result_type = ::std::conditional_t<::std::is_void_v<value_type>, unit, value_type>;
34 using variant_type = ::std::variant<::std::monostate, result_type, ::std::exception_ptr>;
35 struct awaitable_receiver {
36 using receiver_concept = ::beman::execution26::receiver_t;
37
38 template <class... Args>
39 requires ::std::constructible_from<result_type, Args...>
40 void set_value(Args&&... args) && noexcept {
41 try {
42 result_ptr_->template emplace<1>(::std::forward<Args>(args)...);
43 } catch (...) {
44 result_ptr_->template emplace<2>(::std::current_exception());
45 }
46 continuation_.resume();
47 }
48
49 template <class Error>
50 void set_error(Error&& error) && noexcept {
51 result_ptr_->template emplace<2>(
52 ::beman::execution26::detail::as_except_ptr(::std::forward<Error>(error)));
53 continuation_.resume();
54 }
55
56 void set_stopped() && noexcept {
57 static_cast<::std::coroutine_handle<>>(continuation_.promise().unhandled_stopped()).resume();
58 }
59
60 auto get_env() const noexcept {
61 return ::beman::execution26::detail::fwd_env{::beman::execution26::get_env(continuation_.promise())};
62 }
63
64 variant_type* result_ptr_;
65 ::std::coroutine_handle<Promise> continuation_;
66 };
67 using op_state_type = ::beman::execution26::connect_result_t<Sndr, awaitable_receiver>;

69 variant_type result{};

72 public:

74 : state{::beman::execution26::connect(
75 ::std::forward<Sndr>(sndr),

78 static constexpr bool await_ready() noexcept { return false; }
79 void await_suspend(::std::coroutine_handle<Promise>) noexcept { ::beman::execution26::start(state); }
80 value_type await_resume() {

82 ::std::rethrow_exception(::std::get<::std::exception_ptr>(result));

85 return;

89 }
